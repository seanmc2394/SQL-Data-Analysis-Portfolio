Example Query 1:

SELECT 
    books.title,
    SUM(sales.quantity) AS total_copies_sold

FROM 
    sales

JOIN 
    books ON sales.book_id = books.book_id

GROUP BY 
    books.title;




Example Query 2:

SELECT 
    authors.name AS author_name,
    SUM(sales.quantity) AS total_books_sold

FROM 
    sales

JOIN 
    books ON sales.book_id = books.book_id

JOIN 
    authors ON books.author_id = authors.author_id

GROUP BY 
    authors.name

ORDER BY 
    total_books_sold DESC;
